    <script type="module">
        import HesitationEngine from './HesitationEngine.js';

        const btn = document.querySelector('#buy-now');
        
        // הגדרת האלגוריתם על הכפתור
        const engine = new HesitationEngine(btn, {
            exposeDiagnostics: true,
            onUpdate: (state) => {
                // עדכון האלמנטים על המסך בכל פעם שהאלגוריתם פולט חישוב חדש
                document.querySelector('#score-val').textContent = state.hesitation_score;
                document.querySelector('#conf-val').textContent = state.confidence_level;
                
                // תרגום הסיווגים לעברית פשוטה בשביל הבדיקה
                let typeHebrew = "";
                switch(state.predicted_friction_type) {
                    case 'high_intent': typeHebrew = "בדרך לקנייה (בטוח בעצמו)"; break;
                    case 'price_shock': typeHebrew = "מתלבט לגבי המחיר"; break;
                    case 'security_distrust': typeHebrew = "חשדנות / חוסר ביטחון באתר"; break;
                    case 'device_distraction': typeHebrew = "הסחת דעת (עזב את העכבר)"; break;
                    default: typeHebrew = state.predicted_friction_type;
                }
                document.querySelector('#friction-type').textContent = typeHebrew;

                // אקשן אוטומטי לדוגמה: אם המשתמש ממש מהסס לגבי המחיר, נשנה את הכפתור
                if (state.hesitation_score > 70 && state.predicted_friction_type === 'price_shock') {
                    btn.style.backgroundColor = '#2ed573';
                    btn.textContent = 'קנה עכשיו — קבל 3 תשלומים ללא ריבית!';
                }
            }
        });

        // הפעלת המנוע
        engine.start();
    </script>
